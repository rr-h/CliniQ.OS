                <div class="html-embed-6 w-embed w-script">
                  <!DOCTYPE html>
                  <html>

                  <head>
                    <style type="text/css">
                      #game {
                        width: 400px;
                        height: 400px;
                        border: 1px solid #000;
                        margin: auto;
                        position: relative;
                      }

                      #snake {
                        width: 20px;
                        height: 20px;
                        background-color: white;
                        position: absolute;


                      }

                      #food {
                        width: 20px;
                        height: 20px;
                        background-color: #f00;
                        position: absolute;

                      }
                    </style>
                  </head>

                  <body>
                    <div id="game"></div>

                    <script type="text/javascript">
                      // Get the game div
                      var game = document.getElementById('game');

                      // Create the snake
                      var snake = [];
                      snake[0] = {
                        x: 9 * 10,
                        y: 10 * 10
                      };

                      // Create the food
                      var food;

                      // Create the score var
                      var score = 0;

                      // Control the snake
                      var d;

                      // Speed of the game
                      var speed = 100;

                      // Create the canvas
                      function createCanvas() {
                        var canvas = document.createElement('canvas');
                        canvas.width = 400;
                        canvas.height = 400;
                        canvas.style.border = '1px solid #000';
                        game.appendChild(canvas);

                        return canvas.getContext('2d');
                      }

                      // Create the food
                      function createFood() {
                        food = {
                          x: Math.floor(Math.random() * 40) * 10,
                          y: Math.floor(Math.random() * 40) * 10
                        };
                      }

                      // Draw the snake
                      function drawSnake(ctx) {
                        for (var i = 0; i < snake.length; i++) {
                          ctx.fillStyle = (i == 0) ? 'white' : '#f00';
                          ctx.fillRect(snake[i].x, snake[i].y, 10, 10);
                        }
                      }

                      // Draw the food
                      function drawFood(ctx) {
                        ctx.fillStyle = '#f00';
                        ctx.fillRect(food.x, food.y, 10, 10);
                      }

                      // Draw the score
                      function drawScore(ctx) {
                        ctx.fillStyle = 'white';
                        ctx.font = '24px';
                        ctx.fillText('Score: ' + score, 10, 390);
                      }

                      // Draw everything
                      function draw(ctx) {
                        ctx.clearRect(0, 0, 400, 400);

                        drawSnake(ctx);
                        drawFood(ctx);
                        drawScore(ctx);
                      }

                      // Main function
                      function main() {
                        var ctx = createCanvas();

                        createFood();

                        document.addEventListener('keydown', function (e) {
                          if (e.keyCode == 37 && d != 'right') {
                            d = 'left';
                          } else if (e.keyCode == 38 && d != 'down') {
                            d = 'up';
                          } else if (e.keyCode == 39 && d != 'left') {
                            d = 'right';
                          } else if (e.keyCode == 40 && d != 'up') {
                            d = 'down';
                          }
                        });

                        // Call draw function every 100ms
                        setInterval(function () {
                          draw(ctx);

                          // Snake movement
                          var x = snake[0].x;
                          var y = snake[0].y;

                          if (d == 'right') {
                            x += 10;
                          } else if (d == 'left') {
                            x -= 10;
                          } else if (d == 'up') {
                            y -= 10;
                          } else if (d == 'down') {
                            y += 10;
                          }

                          // Check for collision with food
                          if (x == food.x && y == food.y) {
                            score++;
                            createFood();
                          } else {
                            snake.pop();
                          }

                          // Create new head
                          var newHead = {
                            x: x,
                            y: y
                          };

                          // Game over conditions
                          if (x < 0 || x > 420 || y < 0 || y > 420 || collision(newHead, snake)) {

                            document.getElementById("snake-alert").classList.remove("hide");
                            document.getElementById("snake-wrapper").classList.add("hide");
                            document.getElementById("instructions-snake").classList.add("hide");



                          }

                          snake.unshift(newHead);
                        }, speed);
                      }

                      // Check for collision with self
                      function collision(head, array) {
                        for (var i = 0; i < array.length; i++) {
                          if (head.x == array[i].x && head.y == array[i].y) {
                            return true;
                          }
                        }
                        return false;
                      }

                      main();
                    </script>

    function formatSecondsAsTime(secs, format) {
      var hr = Math.floor(secs / 3600);
      var min = Math.floor((secs - (hr * 3600)) / 60);
      var sec = Math.floor(secs - (hr * 3600) - (min * 60));

      if (hr < 10) { hr = "0" + hr; }
      if (min < 10) { min = "0" + min; }
      if (sec < 10) { sec = "0" + sec; }
      if (hr) { hr = "00"; }

      if (format !== null) {
        var formatted_time = format.replace('hh', hr);
        formatted_time = formatted_time.replace('h', hr * 1 + ""); // check for single hour formatting
        formatted_time = formatted_time.replace('mm', min);
        formatted_time = formatted_time.replace('m', min * 1 + ""); // check for single minute formatting
        formatted_time = formatted_time.replace('ss', sec);
        formatted_time = formatted_time.replace('s', sec * 1 + ""); // check for single second formatting
        return formatted_time;
      } else {
        return hr + ':' + min + ':' + sec;
      }
    }

    function setCookie(c_name, value, exdays) {
      var exdate = new Date();
      exdate.setDate(exdate.getDate() + exdays);
      var c_value = escape(value) + ((exdays == null) ? "" : "; expires=" + exdate.toUTCString());
      document.cookie = c_name + "=" + c_value;
    }

    function getCookie(c_name) {
      var i, x, y, ARRcookies = document.cookie.split(";");
      for (i = 0; i < ARRcookies.length; i++) {
        x = ARRcookies[i].substr(0, ARRcookies[i].indexOf("="));
        y = ARRcookies[i].substr(ARRcookies[i].indexOf("=") + 1);
        x = x.replace(/^\s+|\s+$/g, "");
        if (x == c_name) {
          return unescape(y);
        }
      }
    }
